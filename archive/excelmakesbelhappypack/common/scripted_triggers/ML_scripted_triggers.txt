sripted_trigger_loc_dev_cost_less_than = {
	#cheks if value is lower than the devcost
	variable_arithmetic_trigger = {
		export_to_variable = {
			variable_name = varibale	
			value = modifier:local_development_cost
		} #import mod
		NOT = {
			check_variable = {
				which = varibale
				value =  $value$
			}
		}
	}
}
	



is_SPEC_TAG = {
	OR = {
		tag = REB
		tag = NAT
		tag = PIR
}	} 

trigger_of_monarch_power = {
		variable_arithmetic_trigger = {
			export_to_variable = {
				variable_name =  power
				value = trigger_value:$power_type$
			}
			export_to_variable = {
				variable_name = powerCost
				value = modifier:all_power_cost
			}
			change_variable = {
				which = powerCost
				value = 1
			}
			multiply_variable = {
				which = powerCost
				which = 9.99
			}
			subtract_variable = {
				which = monarchCap
				which = powerCost
			}
			check_variable = {
				which = power
				which = monarchCap
			}
		}  
		$power_type$ = $min$
}
