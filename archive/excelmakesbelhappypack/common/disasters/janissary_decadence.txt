janissary_decadence = {
	potential = {
		normal_or_historical_nations = yes
		#government = ottoman_government
		OR = {
			tag = TUR
			has_country_modifier = tur_janissary
		}
	}


	can_start = {
		has_any_disaster = no
		
		OR = {
		AND = {
		has_estate = estate_janissary
		estate_influence = {
			estate = estate_janissary
			influence = 80
		}
		}
		has_country_modifier = tur_janissary
		}
		
		if = {
			limit = { has_dlc = "Cradle of Civilization" }
			if = {
				limit = {
					janissary_percentage = 0.15
					num_of_janissaries = 20
				}
				janissary_percentage = 0.2
				else = {
					custom_trigger_tooltip = {
						tooltip = twenty_percent_janissaries
						janissary_percentage = 0.2
					}
				}
			}
			num_of_janissaries = 20
		}
		
		if = {
			limit = { NOT = { has_dlc = "Cradle of Civilization" } }
			NOT = { adm = 5 }
			NOT = { dip = 5 }
			NOT = { mil = 5 }
		}
		
		
		custom_trigger_tooltip = {
			tooltip = janissary_decadence_trigger
			had_country_flag = {
				flag = janissary_timer
				days = 36500
			}
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = { has_country_modifier = tur_janissary }
			adm = 5
			dip = 5
			mil = 5
		}
		NOT = {
			estate_influence = {
				estate = estate_janissary
				influence = 80
			}		
		}
	}

	
	
	progress = {
		modifier = {
			factor = 2
			if = {
				limit = { has_dlc = "Cradle of Civilization" }
				janissary_percentage = 0.3
				else = { always = no }
			}
		}
		modifier = {
			factor = 1
			if = {
				limit = { has_dlc = "Cradle of Civilization" }
				janissary_percentage = 0.2
				else = { always = no }
			}
		}
		modifier = {
			factor = 0.5
			if = {
				limit = { has_dlc = "Cradle of Civilization" }
				janissary_percentage = 0.1
				else = { always = no }
			}
		}
		modifier = {
			factor = 1
			num_of_loans = 50
		}
		modifier = {
			factor = 1
			is_great_power = no
		}		
		modifier = {
			factor = 1
			has_ruler_modifier = janissaries_denied_reward
		}
		modifier = {
			factor = 2
			is_bankrupt = yes
		}
		modifier = {
			factor = 1
			has_country_modifier = janissary_landholders
		}
		modifier = {
			factor = 1
			has_country_modifier = janissaries_upset
		}

		
		modifier = {
			factor = 1
			estate_influence = {
				estate = estate_janissary
				influence = 80
			}
			hidden_trigger = {
				NOT = {
					estate_influence = {
						estate = estate_janissary
						influence = 85
					}			
				}			
			}
		}
		modifier = {
			factor = 2
			estate_influence = {
				estate = estate_janissary
				influence = 85
			}
			hidden_trigger = {
				NOT = {
					estate_influence = {
						estate = estate_janissary
						influence = 90
					}			
				}			
			}
		}
		modifier = {
			factor = 3
			estate_influence = {
				estate = estate_janissary
				influence = 90
			}
			hidden_trigger = {
				NOT = {
					estate_influence = {
						estate = estate_janissary
						influence = 95
					}			
				}			
			}
		}	
		modifier = {
			factor = 4
			estate_influence = {
				estate = estate_janissary
				influence = 95
			}
			hidden_trigger = {
				NOT = {
					estate_influence = {
						estate = estate_janissary
						influence = 100
					}			
				}			
			}
		}	
		modifier = {
			factor = 5
			estate_influence = {
				estate = estate_janissary
				influence = 100
			}
		}

	}




	
	can_end = {
		NOT = { has_country_modifier = tur_janissary }
		NOT = { has_estate = estate_janissary }
	}
	



	modifier = {
		technology_cost = 0.25
		idea_cost = 0.25
		global_manpower_modifier = -0.25
	}

		
		
		
	
	on_start = janissary_decadence.1
	on_end = janissary_decadence.100
	
	on_monthly = {
		events = {
		}
		
		# about 1 event each 10 years.
		random_events = { 
			1250 = 0
			10 = janissary_decadence.2 #Ruler Killed
			10 = janissary_decadence.3
			10 = janissary_decadence.4
			10 = janissary_decadence.5
			10 = janissary_decadence.6
			10 = janissary_decadence.7
		}
	}
}
