		tag = GEO
		is_subject = yes
		overlord = { 
			has_country_flag = persia_shahanshah_of_persia_march
		}
	}

		if = {

			}
		}

			is_subject_of_type = march
		}
	}
	
	hostile_attrition = 1
	defensiveness = 0.2
}
