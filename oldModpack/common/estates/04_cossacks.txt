estate_cossacks = {
	icon = 4
	
	color = { 200 150 150 }
	
	# Used for when declaring independence
	independence_government = tribal
	starting_reform = steppe_horde
	independence_tech_group = nomad_group
	independence_trigger = {
		OR = { 		
			has_terrain = steppe
			has_estate = estate_cossacks
		}
	}
	
	# Is estate kept in control of province on conquest?
	keep_on_conquest = no
	
	# If true, country will get estate
	
	trigger = {	
		has_dlc = "The Cossacks"
		NOT = { has_reform = pirate_republic_reform }
		NOT = { has_reform = cossacks_reform }
		NOT = { has_government_attribute = is_merchant_republic }
		NOT = { has_reform = trading_city }
		NOT = { has_reform = revolutionary_republic_reform }
		NOT = { government = native }
		NOT = { has_reform = siberian_tribe }
		NOT = { has_reform = veche_republic }
		OR = {
		is_nomad = no
			has_government_attribute = enables_cossack_estate
		}
		is_colonial_nation = no
		NOT = { has_reform = celestial_empire }
		OR = {
		unit_type = eastern
			has_government_attribute = enables_cossack_estate
		}
		OR = {
		religion_group = christian
			has_government_attribute = enables_cossack_estate
		}
	}
	
	# If true, province can be granted to estate
	province_trigger = {
		has_terrain = steppe
	}	
	
	# Giving territory affects power this much compared to base
	territory_power_modifier = 2.0

	# Is used to weigh importance compared to other estates
	start_territory_weight = 100
	
	# Min autonomy in estate provinces
	min_autonomy = 0
	
	# Estate cancels out the following multiplicative local autonomy effects:
	#autonomy_does_not_affect = local_manpower_modifier
	#autonomy_does_not_affect = land_forcelimit_modifier

	# These scale with loyalty & power
	country_modifier_happy = {
		cavalry_power = 0.2
		cavalry_cost = -0.2
		cav_to_inf_ratio = 0.25
	}
	country_modifier_neutral = {
		cavalry_power = 0.1
	}	
	country_modifier_angry = {
		cavalry_cost = 0.2
		cavalry_power = -0.1
	}
	
	# These do not scale, but only applied in provinces owned by the estate
	province_modifier_happy = {	
		local_hostile_attrition = 1
		local_regiment_cost = -0.25
		local_autonomy = -0.1
	}
	province_modifier_neutral = {
		local_autonomy = 0.1
	}	
	province_modifier_angry = {
		local_unrest = 5
		local_autonomy = 0.2	
	}
	
	# Influence modifiers
	base_influence = 0
	influence_modifier = {
		desc = EST_STEPPE_PROVINCES
		trigger = {
			any_owned_province = {
				has_terrain = steppe
			}
			num_of_cities = 10
		}
		influence = 10
	}
	influence_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
				has_disaster = estate_brahmins_disaster
				has_disaster = estate_vaisyas_disaster
				has_disaster = estate_jains_disaster
			}
		}
		influence = -40
	}
	
	loyalty_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER_LOY
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
				has_disaster = estate_brahmins_disaster
				has_disaster = estate_vaisyas_disaster
				has_disaster = estate_jains_disaster
			}
		}
		loyalty = -20
	}
	
	# For province assignment
	ai_grant_province = {
		factor = 10		
		modifier = {
			factor = 4
			local_autonomy = 25
		}
		modifier = {
			factor = 0.25
			NOT = { local_autonomy = 25 }
		}
	}	
	
	# Does the AI want to give any extra territory beyond the minimum?
	ai_territory_modifier = {
		trigger = {
			always = yes
		}	
		value = 100	# Code has a sanity check against pushing influence too high
	}
