estate_dhimmi = {
	icon = 6
	
	color = { 100 125 150 }

	# Used for when declaring independence
	independence_trigger = {

	# Is estate kept in control of province on conquest?
	keep_on_conquest = no	

	# Giving territory affects power this much compared to base
	territory_power_modifier = 0.5	

	# If true, country will get estate
	trigger = {
		has_dlc = "The Cossacks"

		NOT = { has_reform = trading_city }

		NOT = { has_reform = veche_republic }
		NOT = { has_reform = pirate_republic_reform }
		is_nomad = no
		is_colonial_nation = no
		NOT = { has_reform = celestial_empire }
		NOT = { technology_group = western }
		religion_group = muslim
	}
	
	# If true, province can be granted to estate
	province_trigger = {
		NOT = { religion_group = muslim }
		expelling_minorities = no
	}	

	# Is used to weigh importance compared to other estates
	start_territory_weight = 25
	
	# Min autonomy in estate provinces
	min_autonomy = 0
	
	# Estate cancels out the following multiplicative local autonomy effects:
	# autonomy_does_not_affect = local_tax_modifier

	# These scale with loyalty & power
	country_modifier_happy = {
		technology_cost = -0.1
		tolerance_heathen = 2
	}
	country_modifier_neutral = {
		tolerance_heathen = 1
	}	
	country_modifier_angry = {
		technology_cost = 0.1
	}
	
	# These do not scale, but only applied in provinces owned by the estate
	province_modifier_happy = {
		local_tax_modifier = 0.25
		local_unrest = -2
		local_missionary_strength = -1
		local_religious_unity_contribution = -1
		local_autonomy = -0.1
	}
	province_modifier_neutral = {
		local_missionary_strength = -1
		local_religious_unity_contribution = -0.5
		local_autonomy = 0.1
	}		
	province_modifier_angry = {
		local_unrest = 5
		local_missionary_strength = -1	
		local_autonomy = 0.2		
	}	
	
	# Influence modifiers
	base_influence = 0
	influence_modifier = {
		desc = EST_VAL_RELIGIOUS_MINORITY
		trigger = {
			any_owned_province = {
				NOT = { religion_group = muslim }
			}
		}
		influence = 10		
	}
	influence_modifier = {
		desc = EST_VAL_DHIMMI_NOBLES_REFORM
		trigger = {
			has_reform = dhimmi_nobles_reform
		}
		influence = 20
	}

		desc = EST_VAL_RULED_BY_RELIGIOUS_MINORITY
		trigger = {
			NOT = { dominant_religion = ROOT }

				NOT = { religion_group = muslim }
			}
		}
		influence = 20
	}

		desc = EST_VAL_OTHER_ESTATE_IN_POWER

		desc = EST_VAL_FULL_HUMANIST
		trigger = {
			full_idea_group = humanist_ideas
		}
		loyalty = 10
	}

		trigger = {
			OR = {
				technology_group = ottoman
				has_idea_group = humanist_ideas

		}

		factor = 1
		modifier = {
			factor = 10
			has_owner_religion = no

			factor = 4
			local_autonomy = 25
		}
		modifier = {
			factor = 0.25
			NOT = { local_autonomy = 25 }
		}
